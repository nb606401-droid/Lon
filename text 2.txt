local function getNPCByName(name)
    for _, npc in pairs(workspace.NPCs:GetChildren()) do
        if npc:FindFirstChild("Humanoid") and npc.Humanoid.Health > 0 then
            if string.find(npc.Name, name) then
                return npc
            end
        end
    end
    return nil
end

local function getAllBosses()
    local bosses = {}
    for _, name in ipairs(CONFIG.BossList) do
        local boss = getNPCByName(name)
        if boss then table.insert(bosses, boss) end
    end
    return bosses
end

local function getNearestNPC(list)
    local nearest, minDist = nil, math.huge
    for _, name in ipairs(list) do
        local npc = getNPCByName(name)
        if npc then
            local dist = (rootPart.Position - npc.HumanoidRootPart.Position).Magnitude
            if dist < minDist then
                nearest = npc
                minDist = dist
            end
        end
    end
    return nearest
end

local function getNearestBoss()
    local nearest, minDist = nil, math.huge
    local bosses = getAllBosses()
    for _, boss in ipairs(bosses) do
        if boss then
            local dist = (rootPart.Position - boss.HumanoidRootPart.Position).Magnitude
            if dist < minDist then
                nearest = boss
                minDist = dist
            end
        end
    end
    return nearest
end

local function getNearestFruit()
    local closest, minDist = nil, math.huge
    for _, obj in pairs(workspace:GetChildren()) do
        if obj:IsA("Tool") and obj:FindFirstChild("Handle") and string.find(obj.Name, "Fruit") then
            local dist = (rootPart.Position - obj.Handle.Position).Magnitude
            if dist < minDist then
                local fruitName = string.gsub(obj.Name, "Fruit_", "")
                local blocked = false
                for _, black in ipairs(CONFIG.FruitBlacklist) do
                    if string.find(fruitName, black) then blocked = true end
                end
                if not blocked then
                    closest = obj
                    minDist = dist
                end
            end
        end
    end
    return closest
end

local function moveTo(pos)
    tween:Create(rootPart, TweenInfo.new(0.5, Enum.EasingStyle.Linear), {CFrame = CFrame.new(pos)}):Play()
end

local function moveToSmooth(pos)
    local path = pathfindingService:CreatePath()
    path:ComputeAsync(rootPart.Position, pos)
    if path.Status == Enum.PathStatus.Success then
        for _, point in ipairs(path:GetWaypoints()) do
            moveTo(point.Position)
            wait(0.05)
        end
    end
end

local function useKey(key, duration)
    duration = duration or 0.15
    virtualInput:SendKeyEvent(true, key, false, game)
    wait(duration)
    virtualInput:SendKeyEvent(false, key, false, game)
end

local function comboAttack(target)
    if not target or not target:FindFirstChild("HumanoidRootPart") then return end
    local tPos = target.HumanoidRootPart.Position
    if (rootPart.Position - tPos).Magnitude > CONFIG.AttackRange then
        moveTo(tPos + Vector3.new(0, 2, 5))
        wait(0.3)
    end
    rootPart.CFrame = CFrame.new(tPos + Vector3.new(0, 2, 2))
    wait(0.05)
    for _, skill in ipairs(CONFIG.SkillRotation) do
        useKey(skill, 0.08)
        wait(0.1)
    end
end

local function teleportToLocation(location)
    for _, obj in pairs(workspace:GetChildren()) do
        if obj:IsA("Part") and string.find(obj.Name, location) then
            moveTo(obj.Position + Vector3.new(0, 5, 0))
            return true
        end
    end
    return false
end

local espObjects = {}

local function createESP(part, color, text, type)
    if not part then return end
    local bill = Instance.new("BillboardGui")
    bill.Adornee = part
    bill.Size = UDim2.new(0, 200, 0, 50)
    bill.StudsOffset = Vector3.new(0, 3, 0)
    bill.Parent = part
    bill.Name = "ESP_" .. type

    local lbl = Instance.new("TextLabel")
    lbl.Text = text
    lbl.TextColor3 = color
    lbl.Size = UDim2.new(1, 0, 1, 0)
    lbl.BackgroundTransparency = 1
    lbl.TextScaled = true
    lbl.TextStrokeTransparency = 0.3
    lbl.Parent = bill

    if CONFIG.ShowHealth and (type == "Mob" or type == "Boss" or type == "Player") then
        local hpBar = Instance.new("Frame")
        hpBar.Parent = bill
        hpBar.Size = UDim2.new(1, 0, 0, 5)
        hpBar.Position = UDim2.new(0, 0, 1, 2)
        hpBar.BackgroundColor3 = COLORS.Danger
        hpBar.BackgroundTransparency = 0.5

        local hpFill = Instance.new("Frame")
        hpFill.Parent = hpBar
        hpFill.Size = UDim2.new(1, 0, 1, 0)
        hpFill.BackgroundColor3 = COLORS.Success
        hpFill.BackgroundTransparency = 0.3

        local target = part.Parent
        if target and target:FindFirstChild("Humanoid") then
            runService.Heartbeat:Connect(function()
                local hp = target.Humanoid.Health / target.Humanoid.MaxHealth
                hpFill.Size = UDim2.new(hp, 0, 1, 0)
            end)
        end
    end

    if CONFIG.ShowDistance then
        local distLbl = Instance.new("TextLabel")
        distLbl.Parent = bill
        distLbl.Size = UDim2.new(1, 0, 0, 20)
        distLbl.Position = UDim2.new(0, 0, -0.5, 0)
        distLbl.BackgroundTransparency = 1
        distLbl.Text = ""
        distLbl.TextColor3 = COLORS.TextDim
        distLbl.TextSize = 12
        distLbl.Font = Enum.Font.Gotham

        runService.Heartbeat:Connect(function()
            local dist = (rootPart.Position - part.Position).Magnitude
            distLbl.Text = string.format("%.1fm", dist)
        end)
    end

    table.insert(espObjects, bill)
    return bill
end

local function updateESP()
    for _, esp in ipairs(espObjects) do
        if esp and esp.Parent then
            esp:Destroy()
        end
    end
    espObjects = {}

    if CONFIG.ESPPlayers then
        for _, plr in pairs(players:GetPlayers()) do
            if plr ~= player and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                createESP(plr.Character.HumanoidRootPart, COLORS.Primary, "👤 " .. plr.Name, "Player")
            end
        end
    end

    if CONFIG.ESPMobs then
        for _, npc in pairs(workspace.NPCs:GetChildren()) do
            if npc:FindFirstChild("HumanoidRootPart") and npc.Humanoid.Health > 0 then
                local isBoss = false
                for _, name in ipairs(CONFIG.BossList) do
                    if string.find(npc.Name, name) then isBoss = true end
                end
                if not isBoss then
                    createESP(npc.HumanoidRootPart, COLORS.TextDim, "👾 " .. npc.Name, "Mob")
                end
            end
        end
    end

    if CONFIG.ESPBoss then
        for _, name in ipairs(CONFIG.BossList) do
            local boss = getNPCByName(name)
            if boss and boss:FindFirstChild("HumanoidRootPart") then
                createESP(boss.HumanoidRootPart, COLORS.Secondary, "👑 " .. name, "Boss")
            end
        end
    end

    if CONFIG.ESPChest then
        for _, obj in pairs(workspace:GetChildren()) do
            if obj:IsA("Part") and string.find(obj.Name, "Chest") then
                createESP(obj, COLORS.Accent, "📦 Chest", "Chest")
            end
        end
    end

    if CONFIG.ESPFruit then
        for _, obj in pairs(workspace:GetChildren()) do
            if obj:IsA("Tool") and string.find(obj.Name, "Fruit") and obj:FindFirstChild("Handle") then
                local isLegendary = string.find(obj.Name, "Dragon") or string.find(obj.Name, "Leopard") or string.find(obj.Name, "Dough")
                local color = isLegendary and COLORS.Secondary or COLORS.Success
                createESP(obj.Handle, color, "🍎 " .. obj.Name, "Fruit")
            end
        end
    end

    if CONFIG.ESPItems then
        for _, obj in pairs(workspace:GetChildren()) do
            if obj:IsA("Tool") and not string.find(obj.Name, "Fruit") then
                createESP(obj:FindFirstChild("Handle") or obj, COLORS.TextDim, "📦 " .. obj.Name, "Item")
            end
        end
    end
end

        state = not state
        toggleBtn.BackgroundColor3 = state and COLORS.Success or COLORS.Danger
        toggleBtn.Text = state and "ON" or "OFF"
        if callback then callback(state) end
    end)
    return bg
end

        dragStart = input.Position
        startPos = mainFrame.Position
    end

    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)
userInput.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

    if processed then return end

end)
