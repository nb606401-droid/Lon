-- TASK SCHEDULER NÂNG CAO
local function scheduleTask(callback, delay, priority)
    local id = #REDZ.TaskQueue + 1
    REDZ.TaskQueue[id] = {
        callback = callback,
        delay = delay or 0.05,
        priority = priority or 1,
        lastRun = os.clock(),
        active = true,
        runs = 0,
    }
    return id
end

local function processTasks()
    local now = os.clock()
    for id, task in pairs(REDZ.TaskQueue) do
        if task.active and (now - task.lastRun) >= task.delay then
            task.lastRun = now
            task.runs = task.runs + 1
            local success, err = pcall(task.callback)
            if not success then
                REDZ.Stats.Errors = REDZ.Stats.Errors + 1
                if task.runs > 100 then
                    task.active = false
                end
            else
                REDZ.Stats.TasksProcessed = REDZ.Stats.TasksProcessed + 1
            end
        end
    end
end

-- CACHE SYSTEM NÂNG CAO
local function getCachedData(key)
    if REDZ.Cache[key] then
        REDZ.Stats.CacheHits = REDZ.Stats.CacheHits + 1
        return REDZ.Cache[key].data
    end
    REDZ.Stats.CacheMisses = REDZ.Stats.CacheMisses + 1
    return nil
end

local function setCachedData(key, data, ttl)
    REDZ.Cache[key] = {
        data = data,
        expires = os.time() + (ttl or REDZ.Performance.CacheTTL),
    }
end

local function clearCache()
    local now = os.time()
    for key, entry in pairs(REDZ.Cache) do
        if entry.expires < now then
            REDZ.Cache[key] = nil
        end
    end
end

local function getNearestNPC(range, ignoreBoss)
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local nearest, nearestDist = nil, range or State.settings.farmRange
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("Model") and obj:FindFirstChild("Humanoid") and obj:FindFirstChild("HumanoidRootPart") and obj ~= char then
            local hum = obj.Humanoid
            if hum.Health > 0 then
                local isBossMob = false
                if ignoreBoss then isBossMob = isBoss(obj) end
                if not isBossMob then
                    local dist = (obj.HumanoidRootPart.Position - myPos).Magnitude
                    if dist < nearestDist then nearestDist = dist; nearest = obj end
                end
            end
        end
    end
    return nearest
end

local function getNearestBoss(range)
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local nearest, nearestDist = nil, range or State.settings.bossRange
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("Model") and obj:FindFirstChild("Humanoid") and obj:FindFirstChild("HumanoidRootPart") and obj ~= char then
            local hum = obj.Humanoid
            if hum.Health > 0 and isBoss(obj) then
                local dist = (obj.HumanoidRootPart.Position - myPos).Magnitude
                if dist < nearestDist then nearestDist = dist; nearest = obj end
            end
        end
    end
    return nearest
end

local function getNearestFruit(range)
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local nearest, nearestDist = nil, range or 500
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("Tool") and obj:FindFirstChild("Handle") then
            local name = obj.Name:lower()
            if name:find("fruit") or name:find("blox") then
                local dist = (obj.Handle.Position - myPos).Magnitude
                if dist < nearestDist then nearestDist = dist; nearest = obj end
            end
        end
    end
    return nearest
end

local function getNearestChest(range)
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local nearest, nearestDist = nil, range or 300
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj.Name:lower():find("chest") and (obj:IsA("BasePart") or obj:IsA("Model")) then
            local pos = obj:IsA("BasePart") and obj.Position or (obj:FindFirstChild("Handle") and obj.Handle.Position)
            if pos then
                local dist = (pos - myPos).Magnitude
                if dist < nearestDist then nearestDist = dist; nearest = obj end
            end
        end
    end
    return nearest
end

local function getNearestSeaBeast(range)
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local nearest, nearestDist = nil, range or 500
    local seaBeastNames = {"Sea Beast", "Terrorshark", "Leviathan", "Kraken", "Ship Raid", "Pirate Raid", "Volcano", "Sea Serpent"}
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("Model") and obj:FindFirstChild("Humanoid") and obj:FindFirstChild("HumanoidRootPart") and obj ~= char then
            for _, sbn in ipairs(seaBeastNames) do
                if obj.Name:lower():find(sbn:lower()) then
                    local dist = (obj.HumanoidRootPart.Position - myPos).Magnitude
                    if dist < nearestDist then nearestDist = dist; nearest = obj end
                    break
                end
            end
        end
    end
    return nearest
end

local function getNearestPlayer(range)
    local char = LocalPlayer.Character
    if not char or not char:FindFirstChild("HumanoidRootPart") then return nil end
    local myPos = char.HumanoidRootPart.Position
    local nearest, nearestDist = nil, range or 100
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            local hum = plr.Character:FindFirstChildOfClass("Humanoid")
            if hum and hum.Health > 0 then
                local dist = (plr.Character.HumanoidRootPart.Position - myPos).Magnitude
                if dist < nearestDist then nearestDist = dist; nearest = plr end
            end
        end
    end
    return nearest
end

local function equipBestWeapon()
    local char = LocalPlayer.Character
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return end
    local bestTool, bestDamage = nil, 0
    for _, tool in ipairs(LocalPlayer.Backpack:GetChildren()) do
        if tool:IsA("Tool") and tool:FindFirstChild("Handle") then
            local dmg = 0
            pcall(function()
                if tool:FindFirstChild("Config") then
                    dmg = require(tool.Config).Damage or 0
                end
            end)
            if dmg > bestDamage then bestDamage = dmg; bestTool = tool end
        end
    end
    if bestTool then pcall(function() hum:EquipTool(bestTool) end) end
end

-- FLY
local flyBV, flyBG
local flyKeys = {fwd = false, back = false, left = false, right = false, up = false, down = false}
local keyMap = {
    [Enum.KeyCode.W] = "fwd", [Enum.KeyCode.S] = "back",
    [Enum.KeyCode.A] = "left", [Enum.KeyCode.D] = "right",
    [Enum.KeyCode.Space] = "up", [Enum.KeyCode.LeftControl] = "down",
}
local inputBeganConn, inputEndedConn

local function enableFly()
    local char = LocalPlayer.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    flyBV = Instance.new("BodyVelocity")
    flyBV.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    flyBV.Velocity = Vector3.zero
    flyBV.Parent = hrp
    
    flyBG = Instance.new("BodyGyro")
    flyBG.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
    flyBG.P = 9000
    flyBG.Parent = hrp
    
    inputBeganConn = UserInputService.InputBegan:Connect(function(inp, gpe)
        if gpe then return end
        local k = keyMap[inp.KeyCode]; if k then flyKeys[k] = true end
    end)
    inputEndedConn = UserInputService.InputEnded:Connect(function(inp)
        local k = keyMap[inp.KeyCode]; if k then flyKeys[k] = false end
    end)
    
    addConn("flyRender", RunService.RenderStepped:Connect(function()
        if not State.toggles.fly then return end
        local root = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if not root or not flyBV then return end
        local cam = Workspace.CurrentCamera
        local dir = Vector3.zero
        if flyKeys.fwd then dir = dir + cam.CFrame.LookVector end
        if flyKeys.back then dir = dir - cam.CFrame.LookVector end
        if flyKeys.left then dir = dir - cam.CFrame.RightVector end
        if flyKeys.right then dir = dir + cam.CFrame.RightVector end
        if flyKeys.up then dir = dir + Vector3.new(0, 1, 0) end
        if flyKeys.down then dir = dir - Vector3.new(0, 1, 0) end
        if dir.Magnitude > 0 then dir = dir.Unit end
        flyBV.Velocity = dir * 80
        flyBG.CFrame = cam.CFrame
    end))
end
local function disableFly()
    clearConns("flyRender")
    pcall(function() if flyBV then flyBV:Destroy(); flyBV = nil end end)
    pcall(function() if flyBG then flyBG:Destroy(); flyBG = nil end end)
    pcall(function() if inputBeganConn then inputBeganConn:Disconnect() end end)
    pcall(function() if inputEndedConn then inputEndedConn:Disconnect() end end)
    for k in pairs(flyKeys) do flyKeys[k] = false end
end

-- CLICK TP
local function enableClickTP()
    addConn("clickTP", UserInputService.InputBegan:Connect(function(input, gpe)
        if gpe or not State.toggles.clickTP then return end
        if input.UserInputType ~= Enum.UserInputType.MouseButton1 and input.UserInputType ~= Enum.UserInputType.Touch then return end
        local char = LocalPlayer.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then return end
        local pos = input.Position
        local guiAtPos = PlayerGui:GetGuiObjectsAtPosition(pos.X, pos.Y)
        for _, obj in ipairs(guiAtPos) do if obj:IsDescendantOf(ScreenGui) then return end end
        local ray = Camera:ScreenPointToRay(pos.X, pos.Y)
        local params = RaycastParams.new()
        params.FilterDescendantsInstances = {char}
        params.FilterType = Enum.RaycastFilterType.Exclude
        local result = Workspace:Raycast(ray.Origin, ray.Direction * 1000, params)
        if result then
            local targetPos = result.Position + Vector3.new(0, 3, 0)
            char.HumanoidRootPart.CFrame = CFrame.new(targetPos)
        end
    end))
end
local function disableClickTP() clearConns("clickTP") end

local fruitESPDrawings = {}
local legendaryFruits = {"Dragon", "Dough", "Venom", "Spirit", "Control", "Shadow", "Gravity", "Phoenix", "Rumble", "Portal", "Buddha", "Blizzard", "Mammoth", "T-Rex", "Leopard", "Kitsune", "Yeti", "Gas"}
local function enableFruitESP()
    State.bloxFarm.fruitESP = true
    addConn("fruitESPLoop", RunService.RenderStepped:Connect(function()
        if not State.toggles.fruitESP then return end
        for _, d in pairs(fruitESPDrawings) do pcall(function() d:Remove() end) end
        fruitESPDrawings = {}
        for _, obj in ipairs(Workspace:GetDescendants()) do
            if obj:IsA("Tool") and obj:FindFirstChild("Handle") then
                local name = obj.Name
                if name:lower():find("fruit") then
                    local pos, onScreen = Camera:WorldToViewportPoint(obj.Handle.Position)
                    if onScreen then
                        local isLegendary = false
                        for _, lf in ipairs(legendaryFruits) do
                            if name:lower():find(lf:lower()) then isLegendary = true; break end
                        end
                        local txt = Drawing.new("Text")
                        txt.Text = name .. (isLegendary and " ★LEGENDARY★" or "")
                        txt.Position = Vector2.new(pos.X, pos.Y)
                        txt.Size = isLegendary and 18 or 14
                        txt.Color = isLegendary and Color3.fromRGB(255, 215, 0) or Color3.fromRGB(255, 255, 255)
                        txt.Center = true
                        txt.Outline = true
                        txt.OutlineColor = Color3.new(0, 0, 0)
                        txt.Font = 2
                        txt.Visible = true
                        txt.ZIndex = 10
                        table.insert(fruitESPDrawings, txt)
                        local circle = Drawing.new("Circle")
                        circle.Position = Vector2.new(pos.X, pos.Y)
                        circle.Radius = isLegendary and 10 or 6
                        circle.Color = isLegendary and Color3.fromRGB(255, 215, 0) or Theme.success
                        circle.Filled = true
                        circle.Transparency = 0.7
                        circle.Visible = true
                        circle.ZIndex = 9
                        table.insert(fruitESPDrawings, circle)
                    end
                end
            end
        end
    end))
end
local function disableFruitESP()
    State.bloxFarm.fruitESP = false
    clearConns("fruitESPLoop")
    for _, d in pairs(fruitESPDrawings) do pcall(function() d:Remove() end) end
    fruitESPDrawings = {}
end

-- FRUIT RADAR
local fruitRadarDrawings = {}
local function enableFruitRadar()
    addConn("fruitRadarLoop", RunService.RenderStepped:Connect(function()
        if not State.toggles.fruitRadar then return end
        for _, d in pairs(fruitRadarDrawings) do pcall(function() d:Remove() end) end
        fruitRadarDrawings = {}
        local char = LocalPlayer.Character
        if not char or not char:FindFirstChild("HumanoidRootPart") then return end
        local myPos = char.HumanoidRootPart.Position
        for _, obj in ipairs(Workspace:GetDescendants()) do
            if obj:IsA("Tool") and obj:FindFirstChild("Handle") and obj.Name:lower():find("fruit") then
                local pos = obj.Handle.Position
                local dist = (pos - myPos).Magnitude
                if dist < State.settings.espRange then
                    local dir = (pos - myPos).Unit
                    local screenCenter = Vector2.new(viewportSize.X / 2, viewportSize.Y / 2)
                    local angle = math.atan2(dir.Z, dir.X)
                    local radarPos = screenCenter + Vector2.new(math.sin(angle), math.cos(angle)) * 100
                    local txt = Drawing.new("Text")
                    txt.Text = obj.Name .. " [" .. math.floor(dist) .. "m]"
                    txt.Position = radarPos
                    txt.Size = 12
                    txt.Color = Color3.fromRGB(255, 255, 0)
                    txt.Center = true
                    txt.Outline = true
                    txt.Visible = true
                    txt.ZIndex = 10
                    table.insert(fruitRadarDrawings, txt)
                end
            end
        end
    end))
end
local function disableFruitRadar()
    clearConns("fruitRadarLoop")
    for _, d in pairs(fruitRadarDrawings) do pcall(function() d:Remove() end) end
    fruitRadarDrawings = {}
end

-- ESP PLAYERS
local espDrawings = {}
local function createESP(plr)
    if espDrawings[plr.UserId] then return end
    local d = {}
    d.box = Drawing.new("Square")
    d.box.Filled = false; d.box.Color = Theme.accent; d.box.Thickness = 1.5
    d.box.Transparency = 0.9; d.box.Visible = false; d.box.ZIndex = 5
    d.name = Drawing.new("Text")
    d.name.Text = plr.Name; d.name.Size = math.floor(13 * scale); d.name.Center = true
    d.name.Outline = true; d.name.OutlineColor = Color3.new(0, 0, 0)
    d.name.Color = Color3.new(1, 1, 1); d.name.Font = 2
    d.name.Transparency = 1; d.name.Visible = false; d.name.ZIndex = 5
    d.hpBar = Drawing.new("Square")
    d.hpBar.Filled = true; d.hpBar.Color = Theme.success
    d.hpBar.Transparency = 0.8; d.hpBar.Visible = false; d.hpBar.ZIndex = 5
    d.dist = Drawing.new("Text")
    d.dist.Text = ""; d.dist.Size = 10; d.dist.Center = true
    d.dist.Outline = true; d.dist.Color = Color3.new(1, 1, 0)
    d.dist.Visible = false; d.dist.ZIndex = 5
    espDrawings[plr.UserId] = d
end
local function removeESP(plr)
    local d = espDrawings[plr.UserId]
    if d then for _, obj in pairs(d) do pcall(function() obj:Remove() end) end end
    espDrawings[plr.UserId] = nil
end
local function enableESP()
    for _, p in ipairs(Players:GetPlayers()) do if p ~= LocalPlayer then createESP(p) end end
    addConn("espAdd", Players.PlayerAdded:Connect(function(p) if p ~= LocalPlayer then createESP(p) end end))
    addConn("espRem", Players.PlayerRemoving:Connect(removeESP))
    addConn("espUpd", RunService.RenderStepped:Connect(function()
        if not State.toggles.esp then return end
        local char = LocalPlayer.Character
        local myPos = char and char:FindFirstChild("HumanoidRootPart") and char.HumanoidRootPart.Position
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer then
                local d = espDrawings[plr.UserId]
                if d and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") and plr.Character:FindFirstChild("Humanoid") then
                    local hrp = plr.Character.HumanoidRootPart
                    local hum = plr.Character.Humanoid
                    local pos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
                    if onScreen and hum.Health > 0 then
                        local sc = 1 / (pos.Z * 0.026)
                        local w, h = 3.5 * sc, 5 * sc
                        local x, y = pos.X - w / 2, pos.Y - h / 2
                        d.box.Size = Vector2.new(w, h); d.box.Position = Vector2.new(x, y); d.box.Visible = true
                        d.name.Position = Vector2.new(pos.X, y - 16 * scale); d.name.Visible = true
                        local hp = hum.Health / hum.MaxHealth
                        d.hpBar.Size = Vector2.new(3, h * hp); d.hpBar.Position = Vector2.new(x - 6, y + h * (1 - hp))
                        d.hpBar.Color = hp > 0.5 and Theme.success or (hp > 0.25 and Color3.fromRGB(255, 165, 0) or Theme.danger)
                        d.hpBar.Visible = true
                        if myPos and State.toggles.espDistance then
                            local dist = (hrp.Position - myPos).Magnitude
                            d.dist.Text = math.floor(dist) .. "m"
                            d.dist.Position = Vector2.new(pos.X, y + h + 12)
                            d.dist.Visible = true
                        end
                    else
                        d.box.Visible = false; d.name.Visible = false; d.hpBar.Visible = false; d.dist.Visible = false
                    end
                elseif d then
                    d.box.Visible = false; d.name.Visible = false; d.hpBar.Visible = false; d.dist.Visible = false
                end
            end
        end
    end))
end
local function disableESP()
    clearConns("espAdd"); clearConns("espRem"); clearConns("espUpd")
    for _, d in pairs(espDrawings) do for _, o in pairs(d) do pcall(function() o.Visible = false end) end end
end

local function enableChams()

        if not State.toggles.chams then return end
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                for _, part in ipairs(plr.Character:GetDescendants()) do

-- SILENT AIM
local oldNamecall
local function enableSilentAim()
    if oldNamecall then return end
    oldNamecall = hookmetamethod(game, "__namecall", newcclosure(function(self, ...)
        local method = getnamecallmethod()
        if State.toggles.silentAim and method == "Raycast" then
            local args = {...}
            local t = getClosestPlayer()
            if t and t.Character and t.Character:FindFirstChild("Head") then
                local head = t.Character.Head
                local origin = args[1] or Camera.CFrame.Position
                if #args >= 2 then args[2] = (head.Position - origin) end
                return oldNamecall(self, unpack(args))
            end
        end
        return oldNamecall(self, ...)
    end))
end
local function disableSilentAim()
    if oldNamecall then
        pcall(function() hookmetamethod(game, "__namecall", oldNamecall); oldNamecall = nil end)
    end
end

function applyToggle(key)
    local h = handlers[key]
    if not h then
        warn("[REDZ] No handler found for key: " .. key)
        return
    end
    if State.toggles[key] then
        h[1]()
        print("[REDZ] Enabled: " .. key)
    else
        h[2]()
        print("[REDZ] Disabled: " .. key)
    end
end

function toggleFeature(key)
    if key == "farmRange" then
        State.settings.farmRange = State.settings.farmRange + 50
        if State.settings.farmRange > 1000 then State.settings.farmRange = 50 end
        setToggleVisual(key, true)
        return
    elseif key == "bossRange" then
        State.settings.bossRange = State.settings.bossRange + 100
        if State.settings.bossRange > 1000 then State.settings.bossRange = 100 end
        setToggleVisual(key, true)
        return
    elseif key == "killAuraRange" then
        State.settings.killAuraRange = State.settings.killAuraRange + 25
        if State.settings.killAuraRange > 500 then State.settings.killAuraRange = 25 end
        setToggleVisual(key, true)
        return
    elseif key == "killAuraDamage" then
        State.settings.killAuraDamage = State.settings.killAuraDamage + 100
        if State.settings.killAuraDamage > 9999 then State.settings.killAuraDamage = 100 end
        setToggleVisual(key, true)
        return
    elseif key == "serverHopThreshold" then
        State.settings.serverHopThreshold = State.settings.serverHopThreshold + 5
        if State.settings.serverHopThreshold > 50 then State.settings.serverHopThreshold = 1 end
        setToggleVisual(key, true)
        return
    elseif key == "farmDelay" then
        State.settings.farmDelay = State.settings.farmDelay + 0.02
        if State.settings.farmDelay > 0.2 then State.settings.farmDelay = 0.02 end
        setToggleVisual(key, true)
        return
    elseif key == "teleportSpeed" then
        State.settings.teleportSpeed = State.settings.teleportSpeed + 50
        if State.settings.teleportSpeed > 1000 then State.settings.teleportSpeed = 100 end
        setToggleVisual(key, true)
        return
    elseif key == "espRange" then
        State.settings.espRange = State.settings.espRange + 100
        if State.settings.espRange > 2000 then State.settings.espRange = 100 end
        setToggleVisual(key, true)
        return
    elseif key == "fruitSniperDelay" then
        State.settings.fruitSniperDelay = State.settings.fruitSniperDelay + 0.5
        if State.settings.fruitSniperDelay > 5 then State.settings.fruitSniperDelay = 0.5 end
        setToggleVisual(key, true)
        return
    elseif key == "seaEventsInterval" then
        State.settings.seaEventsCheckInterval = State.settings.seaEventsCheckInterval + 5
        if State.settings.seaEventsCheckInterval > 60 then State.settings.seaEventsCheckInterval = 5 end
        setToggleVisual(key, true)
        return
    elseif key == "renderInterval" then
        State.settings.renderInterval = State.settings.renderInterval + 0.01
        if State.settings.renderInterval > 0.1 then State.settings.renderInterval = 0.01 end
        REDZ.Performance.RenderInterval = State.settings.renderInterval
        setToggleVisual(key, true)
        return
    elseif key == "optimizationLevel" then
        State.settings.optimizationLevel = State.settings.optimizationLevel + 1
        if State.settings.optimizationLevel > 5 then State.settings.optimizationLevel = 1 end
        REDZ.Performance.OptimizationLevel = State.settings.optimizationLevel
        setToggleVisual(key, true)
        return
    elseif key == "autoFarmSpeed" then
        State.settings.autoFarmSpeed = State.settings.autoFarmSpeed + 1
        if State.settings.autoFarmSpeed > 10 then State.settings.autoFarmSpeed = 1 end
        setToggleVisual(key, true)
        return
    elseif key == "maxFPS" then
        REDZ.Performance.MaxFPS = REDZ.Performance.MaxFPS + 10
        if REDZ.Performance.MaxFPS > 120 then REDZ.Performance.MaxFPS = 30 end
        setToggleVisual(key, true)
        return
    end

    local val = not State.toggles[key]
    State.toggles[key] = val
    State.bloxFarm[key] = val
    setToggleVisual(key, val)
    applyToggle(key)
end

-- ============================================================
-- DRAG
-- ============================================================
HeaderFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        State.dragging = true
        State.dragStart = Vector2.new(input.Position.X, input.Position.Y)
        State.dragStartPos = Vector2.new(State.menuX, State.menuY)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if State.dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = Vector2.new(input.Position.X - State.dragStart.X, input.Position.Y - State.dragStart.Y)
        State.menuX = State.dragStartPos.X + delta.X
        State.menuY = State.dragStartPos.Y + delta.Y
        State.menuX = math.clamp(State.menuX, 0, viewportSize.X - W)
        State.menuY = math.clamp(State.menuY, 0, viewportSize.Y - HEADER_H)
        HeaderFrame.Position = UDim2.new(0, State.menuX, 0, State.menuY)
        ContentFrame.Position = UDim2.new(0, State.menuX, 0, State.menuY + HEADER_H)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        State.dragging = false
    end
end)

UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe then return end

        setMinimized(not State.minimized)
    end
end)
