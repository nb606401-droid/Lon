-- ESP Function
local espHighlights = {}
local function applyESP(target, color)
    if target:FindFirstChild("Highlight") then return end
    local highlight = Instance.new("Highlight")
    highlight.Name = "Highlight"
    highlight.FillColor = color
    highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
    highlight.FillTransparency = 0.5
    highlight.OutlineTransparency = 0
    highlight.Parent = target
    table.insert(espHighlights, highlight)
end

local function updateESP()
    for _, highlight in pairs(espHighlights) do
        if highlight and highlight.Parent then highlight:Destroy() end
    end
    espHighlights = {}

    if not espEnabled then return end

    -- ESP Enemies
    for _, enemy in pairs(workspace.Enemies:GetChildren()) do
        if enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 then
            applyESP(enemy, Color3.fromRGB(255, 0, 0))
        end
    end

    -- ESP Players
    for _, plr in pairs(game.Players:GetPlayers()) do
        if plr \~= player and plr.Character then
            applyESP(plr.Character, Color3.fromRGB(0, 255, 0))
        end
    end
end

-- Main Loop
spawn(function()
    while true do
        wait(0.05)
        local character = player.Character
        if not character then continue end
        local root = character:FindFirstChild("HumanoidRootPart")
        if not root then continue end

        -- Auto Quay Trái
        if autoQuayTrai then
            root.CFrame = root.CFrame * CFrame.Angles(0, math.rad(-8), 0)
        end

        -- ESP
        if espEnabled then
            updateESP()
        end

        -- Auto Farm Level
        if farming then
            for _, enemy in pairs(workspace.Enemies:GetChildren()) do
                if enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 and enemy:FindFirstChild("HumanoidRootPart") then
                    root.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
                    local tool = character:FindFirstChildOfClass("Tool")
                    if tool then tool:Activate() end
                    break
                end
            end
        end

        -- Auto Fruit Sniper
        if fruitSniping then
            for _, obj in pairs(workspace:GetChildren()) do
                if obj:FindFirstChild("Handle") and obj.Name:find("Fruit") then
                    root.CFrame = obj.Handle.CFrame
                    wait(0.4)
                end
            end
        end

        -- Auto Equip Sword
        if autoSword then
            for _, item in pairs(player.Backpack:GetChildren()) do
                if item:IsA("Tool") and (item.Name:lower():find("sword") or item.Name:lower():find("katana")) then
                    if not character:FindFirstChildOfClass("Tool") then
                        item.Parent = character
                    end
                    break
                end
            end
        end

        -- Auto Bounty
        if bountyHunting then
            for _, plr in pairs(game.Players:GetPlayers()) do
                if plr \~= player and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                    root.CFrame = plr.Character.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
                    local tool = character:FindFirstChildOfClass("Tool")
                    if tool then tool:Activate() end
                    break
                end
            end
        end
    end
end)
