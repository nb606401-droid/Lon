-- ================== FLY ==================
local flyBody, flyGyro, flying = nil, nil, false

local function StartFly()
   local char = game.Players.LocalPlayer.Character
   if not char then return end
   local root = char:FindFirstChild("HumanoidRootPart")
   if not root then return end

   if flyBody then flyBody:Destroy() end
   if flyGyro then flyGyro:Destroy() end

   flyBody = Instance.new("BodyVelocity")
   flyBody.MaxForce = Vector3.new(1e5, 1e5, 1e5)
   flyBody.Parent = root

   flyGyro = Instance.new("BodyGyro")
   flyGyro.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
   flyGyro.P = 3000
   flyGyro.Parent = root

   flying = true
   task.spawn(function()
      while flying and root.Parent do
         pcall(function()
            local y = root.Position.Y
            if y < getgenv().FlyHeight - 10 then
               flyBody.Velocity = Vector3.new(0, 95, 0)
            elseif y > getgenv().FlyHeight + 10 then
               flyBody.Velocity = Vector3.new(0, -85, 0)
            else
               flyBody.Velocity = Vector3.new(0, 28, 0)
            end
         end)
         task.wait(0.02)
      end
   end)
end

local function StopFly()
   flying = false
   if flyBody then flyBody:Destroy() end
   if flyGyro then flyGyro:Destroy() end
end

-- ================== HITBOX ==================
local originalSizes = {}
local function EnableHitbox(size)
   for _, enemy in pairs(game.Workspace.Enemies:GetChildren()) do
      local root = enemy:FindFirstChild("HumanoidRootPart")
      if root then
         if not originalSizes[enemy] then originalSizes[enemy] = root.Size end
         root.Size = Vector3.new(size, size, size)
         root.Transparency = 0.5
         root.CanCollide = false
      end
   end
end

local function DisableHitbox()
   for enemy, size in pairs(originalSizes) do
      if enemy and enemy:FindFirstChild("HumanoidRootPart") then
         enemy.HumanoidRootPart.Size = size
         enemy.HumanoidRootPart.Transparency = 1
      end
   end
   originalSizes = {}
end
